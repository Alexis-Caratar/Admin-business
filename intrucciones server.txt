DROP TRIGGER IF EXISTS trg_descontar_inventario ON ventas_items;

CREATE TRIGGER trg_descontar_inventario
AFTER INSERT ON ventas_items
FOR EACH ROW
EXECUTE FUNCTION descontar_inventario();



CREATE OR REPLACE FUNCTION public.descontar_inventario()
RETURNS trigger
LANGUAGE plpgsql
AS $function$
DECLARE
  r RECORD;
  v_usa_receta BOOLEAN;
  v_inventario_id INT;
  v_stock_actual NUMERIC;
  v_controla_inventario BOOLEAN;
BEGIN

  -- 🔍 Obtener configuración del producto
  SELECT usa_receta, inventario_id, controla_inventario
  INTO v_usa_receta, v_inventario_id, v_controla_inventario
  FROM productos
  WHERE id = NEW.id_producto;

  -- 🚨 Validación
  IF v_controla_inventario IS NULL THEN
    RAISE EXCEPTION 'Producto % no configurado correctamente', NEW.id_producto;
  END IF;

  -- ================================
  -- ⚪ PRODUCTO SIN CONTROL DE INVENTARIO
  -- ================================
  IF v_controla_inventario = FALSE THEN
    RETURN NEW;
  END IF;

  -- ================================
  -- 🟢 PRODUCTO CON RECETA
  -- ================================
  IF v_usa_receta = TRUE THEN

    IF NOT EXISTS (
      SELECT 1 FROM recetas WHERE producto_id = NEW.id_producto
    ) THEN
      RAISE EXCEPTION 'El producto % no tiene receta', NEW.id_producto;
    END IF;

    FOR r IN
      SELECT * FROM recetas WHERE producto_id = NEW.id_producto
    LOOP

      SELECT stock_actual INTO v_stock_actual
      FROM inventario
      WHERE id = r.inventario_id;

      IF v_stock_actual < (r.cantidad * NEW.cantidad) THEN
        RAISE EXCEPTION 
        'Stock insuficiente en insumo % (disp: %, req: %)',
        r.inventario_id,
        v_stock_actual,
        (r.cantidad * NEW.cantidad);
      END IF;

      UPDATE inventario
      SET stock_actual = stock_actual - (r.cantidad * NEW.cantidad)
      WHERE id = r.inventario_id;

    END LOOP;

  ELSE
  -- ================================
  -- 🔵 PRODUCTO FÍSICO
  -- ================================

    IF v_inventario_id IS NULL THEN
      RAISE EXCEPTION 'Producto % sin inventario asociado', NEW.id_producto;
    END IF;

    SELECT stock_actual INTO v_stock_actual
    FROM inventario
    WHERE id = v_inventario_id;

    IF v_stock_actual < NEW.cantidad THEN
      RAISE EXCEPTION 
      'Stock insuficiente (disp: %, req: %)',
      v_stock_actual,
      NEW.cantidad;
    END IF;

    UPDATE inventario
    SET stock_actual = stock_actual - NEW.cantidad
    WHERE id = v_inventario_id;

  END IF;

  RETURN NEW;
END;
$function$;
