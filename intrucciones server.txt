CREATE TABLE inventario_movimientos (
  id SERIAL PRIMARY KEY,
  inventario_id INT NOT NULL,
  tipo VARCHAR(10) NOT NULL, -- ENTRADA | SALIDA
  cantidad NUMERIC(10,2) NOT NULL,
  motivo VARCHAR(100),
  referencia VARCHAR(100), -- compra, venta, ajuste
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT fk_inventario
    FOREIGN KEY (inventario_id)
    REFERENCES inventario(id)
    ON DELETE CASCADE
);



DROP TRIGGER IF EXISTS trg_descontar_inventario ON ventas_items;

CREATE TRIGGER trg_descontar_inventario
AFTER INSERT ON ventas_items
FOR EACH ROW
EXECUTE FUNCTION descontar_inventario();


-- DROP FUNCTION public.descontar_inventario();

CREATE OR REPLACE FUNCTION public.descontar_inventario()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
DECLARE
  r RECORD;
  v_usa_receta BOOLEAN;
  v_inventario_id INT;
  v_stock_actual NUMERIC;
BEGIN

  -- 🔍 Obtener configuración del producto
  SELECT usa_receta, inventario_id
  INTO v_usa_receta, v_inventario_id
  FROM productos
  WHERE id = NEW.id_producto;

  -- 🚨 Validación
  IF v_usa_receta IS NULL THEN
    RAISE EXCEPTION 'Producto % no configurado', NEW.id_producto;
  END IF;

  -- ================================
  -- 🟢 PRODUCTO CON RECETA
  -- ================================
  IF v_usa_receta = TRUE THEN

    -- validar que tenga receta
    IF NOT EXISTS (
      SELECT 1 FROM recetas WHERE producto_id = NEW.id_producto
    ) THEN
      RAISE EXCEPTION 'El producto % no tiene receta', NEW.id_producto;
    END IF;

    FOR r IN
      SELECT * FROM recetas WHERE producto_id = NEW.id_producto
    LOOP

      SELECT stock_actual INTO v_stock_actual
      FROM inventario
      WHERE id = r.inventario_id;

      IF v_stock_actual < (r.cantidad * NEW.cantidad) THEN
        RAISE EXCEPTION 
        'stock_actual insuficiente en insumo % (disp: %, req: %)',
        r.inventario_id,
        v_stock_actual,
        (r.cantidad * NEW.cantidad);
      END IF;

      UPDATE inventario
      SET stock_actual = stock_actual - (r.cantidad * NEW.cantidad)
      WHERE id = r.inventario_id;

    END LOOP;

  ELSE
  -- ================================
  -- 🔵 PRODUCTO FÍSICO
  -- ================================

    IF v_inventario_id IS NULL THEN
      RAISE EXCEPTION 'Producto % sin inventario asociado', NEW.id_producto;
    END IF;

    SELECT stock_actual INTO v_stock_actual
    FROM inventario
    WHERE id = v_inventario_id;

    IF v_stock_actual < NEW.cantidad THEN
      RAISE EXCEPTION 
      'stock_actual insuficiente (disp: %, req: %)',
      v_stock_actual,
      NEW.cantidad;
    END IF;

    UPDATE inventario
    SET stock_actual = stock_actual - NEW.cantidad
    WHERE id = v_inventario_id;

  END IF;

  RETURN NEW;
END;
$function$
;
